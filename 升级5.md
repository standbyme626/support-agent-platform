```python
    "timeout",

    status: TicketStatus
    handoff_state: HandoffState
    lifecycle_stage: str
    approval_state: ApprovalState = "none"

    def can_resolve(self) -> bool:
        return self.status not in ("resolved", "closed")

    def can_customer_confirm(self) -> bool:
        return self.status == "resolved" and self.handoff_state == "waiting_customer"

    def can_operator_close(self) -> bool:
        return self.status in ("open", "pending", "escalated", "handoff", "resolved")

    def can_mutate(self) -> bool:
        return self.status != "closed"
```

```python
from app.domain.conversation.conversation_state import ConversationState


class SessionContextStoreProtocol:
    def get(self, session_id: str) -> ConversationState | None: ...
    def save(self, state: ConversationState) -> None: ...


class SessionService:
    def __init__(self, session_store: SessionContextStoreProtocol) -> None:
        self._session_store = session_store

    def get_or_create(self, session_id: str) -> ConversationState:
        state = self._session_store.get(session_id)
        if state is not None:
            return state
        state = ConversationState(session_id=session_id)
        self._session_store.save(state)
        return state

    def set_active_ticket(self, session_id: str, ticket_id: str) -> ConversationState:
        state = self.get_or_create(session_id)
        state.set_active_ticket(ticket_id)
        self._session_store.save(state)
        return state

    def clear_active_ticket(self, session_id: str) -> ConversationState:
        state = self.get_or_create(session_id)
        state.clear_active_ticket()
        self._session_store.save(state)
        return state

    def mark_waiting_customer(
        self,
        session_id: str,
        ticket_id: str,
        flag: bool,
    ) -> ConversationState:
        state = self.get_or_create(session_id)
        if state.active_ticket_id != ticket_id:
            state.set_active_ticket(ticket_id)
        state.mark_waiting_customer(flag)
        self._session_store.save(state)
        return state

    def set_mode(self, session_id: str, mode: str) -> ConversationState:
        state = self.get_or_create(session_id)
        state.conversation_mode = mode  # 可后续收紧为 enum 校验
        self._session_store.save(state)
        return state

    def set_last_intent(self, session_id: str, intent: str) -> ConversationState:
        state = self.get_or_create(session_id)
        state.last_user_intent = intent
        self._session_store.save(state)
        return state

    def end_session(self, session_id: str) -> ConversationState:
        state = self.get_or_create(session_id)
        state.clear_active_ticket()
        state.conversation_mode = "idle"
        self._session_store.save(state)
        return state
```

```python
from __future__ import annotations

from dataclasses import dataclass
from typing import Any

from app.application.session_service import SessionService
from app.domain.ticket.ticket_workflow_state import TicketWorkflowState


@dataclass
class TicketActionResult:
    ticket_id: str
    status: str
    handoff_state: str
    event_type: str
    message: str


class TicketRepositoryProtocol:
    def get_workflow_state(self, ticket_id: str) -> TicketWorkflowState: ...
    def update_ticket_fields(self, ticket_id: str, fields: dict[str, Any]) -> None: ...
    def append_event(self, ticket_id: str, event_type: str, payload: dict[str, Any]) -> None: ...


class TicketAPI:
    def __init__(
        self,
        ticket_repo: TicketRepositoryProtocol,
        session_service: SessionService,
    ) -> None:
        self._ticket_repo = ticket_repo
        self._session_service = session_service

    def resolve_ticket(
        self,
        ticket_id: str,
        actor: str,
        *,
        session_id: str | None = None,
        note: str | None = None,
    ) -> TicketActionResult:
        state = self._ticket_repo.get_workflow_state(ticket_id)
        if not state.can_resolve():
            raise ValueError(f"ticket {ticket_id} cannot be resolved from {state.status}")

        self._ticket_repo.update_ticket_fields(
            ticket_id,
            {
                "status": "resolved",
                "handoff_state": "waiting_customer",
            },
        )
        self._ticket_repo.append_event(
            ticket_id,
            "ticket_resolved",
            {"actor": actor, "note": note},
        )

        if session_id:
            self._session_service.mark_waiting_customer(session_id, ticket_id, True)

        return TicketActionResult(
            ticket_id=ticket_id,
            status="resolved",
            handoff_state="waiting_customer",
            event_type="ticket_resolved",
            message="Ticket resolved and waiting for customer confirmation.",
        )

    def customer_confirm_ticket(
        self,
        ticket_id: str,
        actor: str,
        *,
        session_id: str | None = None,
        note: str | None = None,
    ) -> TicketActionResult:
        state = self._ticket_repo.get_workflow_state(ticket_id)
        if not state.can_customer_confirm():
            raise ValueError(f"ticket {ticket_id} cannot be customer-confirmed")

        self._ticket_repo.update_ticket_fields(
            ticket_id,
            {
                "status": "closed",
                "handoff_state": "completed",
            },
        )
        self._ticket_repo.append_event(
            ticket_id,
            "ticket_customer_confirmed",
            {"actor": actor, "note": note},
        )

        if session_id:
            self._session_service.clear_active_ticket(session_id)

        return TicketActionResult(
            ticket_id=ticket_id,
            status="closed",
            handoff_state="completed",
            event_type="ticket_customer_confirmed",
            message="Ticket closed after customer confirmation.",
        )

    def operator_close_ticket(
        self,
        ticket_id: str,
        actor: str,
        reason: str,
        *,
        session_id: str | None = None,
        note: str | None = None,
    ) -> TicketActionResult:
        state = self._ticket_repo.get_workflow_state(ticket_id)
        if not state.can_operator_close():
            raise ValueError(f"ticket {ticket_id} cannot be operator-closed")

        self._ticket_repo.update_ticket_fields(
            ticket_id,
            {
                "status": "closed",
                "handoff_state": "completed",
                "close_reason": reason,
            },
        )
        self._ticket_repo.append_event(
            ticket_id,
            "ticket_operator_closed",
            {"actor": actor, "reason": reason, "note": note},
        )

        if session_id:
            self._session_service.clear_active_ticket(session_id)

        return TicketActionResult(
            ticket_id=ticket_id,
            status="closed",
            handoff_state="completed",
            event_type="ticket_operator_closed",
            message="Ticket forcibly closed by operator.",
        )

    def end_session(
        self,
        session_id: str,
        actor: str,
        *,
        reason: str | None = None,
    ) -> dict[str, str]:
        self._session_service.end_session(session_id)
        return {
            "session_id": session_id,
            "actor": actor,
            "reason": reason or "manual_end",
            "message": "Session ended successfully.",
        }
```

```python
from __future__ import annotations

from dataclasses import dataclass
from typing import Any


@dataclass
class IntakeInput:
    session_id: str
    user_id: str
    text: str
    channel: str
    metadata: dict[str, Any]


@dataclass
class IntakeDecision:
    intent: str
    reply_text: str
    should_create_or_update_ticket: bool
    ticket_action: str  # create | update | none
    should_push_to_collab: bool
    trace_meta: dict[str, Any]


class IntentRouterProtocol:
    def classify(self, text: str, metadata: dict[str, Any]) -> str: ...


class RetrieverProtocol:
    def search(self, query: str, *, session_id: str) -> list[dict[str, Any]]: ...


class SummaryEngineProtocol:
    def summarize(self, text: str, docs: list[dict[str, Any]]) -> str: ...


class TicketAPIProtocol:
    def create_or_update_from_intake(self, payload: dict[str, Any]) -> dict[str, Any]: ...


class IntakeService:
    def __init__(
        self,
        intent_router: IntentRouterProtocol,
        retriever: RetrieverProtocol,
        summary_engine: SummaryEngineProtocol,
        ticket_api: TicketAPIProtocol,
    ) -> None:
        self._intent_router = intent_router
        self._retriever = retriever
        self._summary_engine = summary_engine
        self._ticket_api = ticket_api

    def classify_intent(self, intake: IntakeInput) -> str:
        return self._intent_router.classify(intake.text, intake.metadata)

    def retrieve_context(self, intake: IntakeInput) -> list[dict[str, Any]]:
        return self._retriever.search(intake.text, session_id=intake.session_id)

    def summarize_context(
        self,
        intake: IntakeInput,
        docs: list[dict[str, Any]],
    ) -> str:
        return self._summary_engine.summarize(intake.text, docs)

    def decide_ticket_action(
        self,
        intake: IntakeInput,
        intent: str,
        summary: str,
        docs: list[dict[str, Any]],
    ) -> IntakeDecision:
        # 这里只放最小骨架，具体规则后续补
        should_create = intent not in ("faq", "smalltalk")
        return IntakeDecision(
            intent=intent,
            reply_text=summary,
            should_create_or_update_ticket=should_create,
            ticket_action="create" if should_create else "none",
            should_push_to_collab=should_create,
            trace_meta={"doc_count": len(docs)},
        )

    def apply_intake_outcome(
        self,
        intake: IntakeInput,
        decision: IntakeDecision,
    ) -> dict[str, Any]:
        if not decision.should_create_or_update_ticket:
            return {"ticket": None, "reply": decision.reply_text}

        ticket = self._ticket_api.create_or_update_from_intake(
            {
                "session_id": intake.session_id,
                "user_id": intake.user_id,
                "text": intake.text,
                "intent": decision.intent,
                "reply_text": decision.reply_text,
                "trace_meta": decision.trace_meta,
            }
        )
        return {"ticket": ticket, "reply": decision.reply_text}
```

```python
from __future__ import annotations

from dataclasses import dataclass
from typing import Any


@dataclass
class CollabCommand:
    ticket_id: str
    action: str
    actor: str
    session_id: str | None = None
    args: dict[str, Any] | None = None


@dataclass
class RiskEvaluation:
    action: str
    needs_approval: bool
    reason: str | None = None


class ApprovalPolicyProtocol:
    def evaluate(self, action: str, args: dict[str, Any] | None) -> RiskEvaluation: ...


class TicketAPIProtocol:
    def resolve_ticket(self, *args, **kwargs): ...
    def customer_confirm_ticket(self, *args, **kwargs): ...
    def operator_close_ticket(self, *args, **kwargs): ...


class CollabService:
    def __init__(
        self,
        approval_policy: ApprovalPolicyProtocol,
        ticket_api: TicketAPIProtocol,
    ) -> None:
        self._approval_policy = approval_policy
        self._ticket_api = ticket_api

    def parse_command(
        self,
        ticket_id: str,
        action: str,
        actor: str,
        *,
        session_id: str | None = None,
        args: dict[str, Any] | None = None,
    ) -> CollabCommand:
        return CollabCommand(
            ticket_id=ticket_id,
            action=action,
            actor=actor,
            session_id=session_id,
            args=args or {},
        )

    def evaluate_risk(self, cmd: CollabCommand) -> RiskEvaluation:
        return self._approval_policy.evaluate(cmd.action, cmd.args)

    def apply_collab_action(self, cmd: CollabCommand) -> dict[str, Any]:
        if cmd.action == "resolve":
            return self._ticket_api.resolve_ticket(
                cmd.ticket_id,
                cmd.actor,
                session_id=cmd.session_id,
                note=(cmd.args or {}).get("note"),
            ).__dict__

        if cmd.action == "customer_confirm":
            return self._ticket_api.customer_confirm_ticket(
                cmd.ticket_id,
                cmd.actor,
                session_id=cmd.session_id,
                note=(cmd.args or {}).get("note"),
            ).__dict__

        if cmd.action == "operator_close":
            return self._ticket_api.operator_close_ticket(
                cmd.ticket_id,
                cmd.actor,
                reason=(cmd.args or {}).get("reason", "manual_close"),
                session_id=cmd.session_id,
                note=(cmd.args or {}).get("note"),
            ).__dict__

        raise NotImplementedError(f"unsupported action: {cmd.action}")
```

```python
from __future__ import annotations

from typing import TypedDict, Any

from langgraph.func import entrypoint, task

from app.application.intake_service import IntakeInput, IntakeService


class IntakeGraphState(TypedDict):
    session_id: str
    user_id: str
    text: str
    channel: str
    metadata: dict[str, Any]


def build_intake_graph(intake_service: IntakeService, checkpointer):
    @task
    def classify_intent_task(intake: IntakeInput) -> str:
        return intake_service.classify_intent(intake)

    @task
    def retrieve_context_task(intake: IntakeInput) -> list[dict[str, Any]]:
        return intake_service.retrieve_context(intake)

    @task
    def summarize_context_task(
        intake: IntakeInput,
        docs: list[dict[str, Any]],
    ) -> str:
        return intake_service.summarize_context(intake, docs)

    @task
    def decide_ticket_action_task(
        intake: IntakeInput,
        intent: str,
        summary: str,
        docs: list[dict[str, Any]],
    ):
        return intake_service.decide_ticket_action(intake, intent, summary, docs)

    @task
    def apply_outcome_task(intake: IntakeInput, decision):
        return intake_service.apply_intake_outcome(intake, decision)

    @entrypoint(checkpointer=checkpointer)
    def run_intake(payload: IntakeGraphState, *, previous=None):
        intake = IntakeInput(
            session_id=payload["session_id"],
            user_id=payload["user_id"],
            text=payload["text"],
            channel=payload["channel"],
            metadata=payload.get("metadata", {}),
        )
        intent = classify_intent_task(intake).result()
        docs = retrieve_context_task(intake).result()
        summary = summarize_context_task(intake, docs).result()
        decision = decide_ticket_action_task(intake, intent, summary, docs).result()
        return apply_outcome_task(intake, decision).result()

    return run_intake
```

```python
from __future__ import annotations

from typing import TypedDict, Any

from langgraph.graph import StateGraph, START, END
from langgraph.types import interrupt, Command

from app.application.collab_service import CollabService


class CollabGraphState(TypedDict, total=False):
    ticket_id: str
    action: str
    actor: str
    session_id: str | None
    args: dict[str, Any]
    needs_approval: bool
    approval_reason: str | None
    result: dict[str, Any]


def build_collab_graph(collab_service: CollabService):
    graph = StateGraph(CollabGraphState)

    def parse_command_node(state: CollabGraphState) -> CollabGraphState:
        cmd = collab_service.parse_command(
            ticket_id=state["ticket_id"],
            action=state["action"],
            actor=state["actor"],
            session_id=state.get("session_id"),
            args=state.get("args", {}),
        )
        return {
            **state,
            "_cmd": cmd,  # 可后续换成更干净的内部字段
        }

    def risk_check_node(state: CollabGraphState) -> CollabGraphState:
        cmd = state["_cmd"]
        risk = collab_service.evaluate_risk(cmd)
        return {
            **state,
            "needs_approval": risk.needs_approval,
            "approval_reason": risk.reason,
        }

    def approval_node(state: CollabGraphState):
        if not state.get("needs_approval"):
            return state

        decision = interrupt(
            {
                "ticket_id": state["ticket_id"],
                "action": state["action"],
                "reason": state.get("approval_reason"),
            }
        )

        if decision == "approved":
            return state

        return Command(
            goto="finalize_rejected_node",
            update={
                "result": {
                    "status": "rejected",
                    "ticket_id": state["ticket_id"],
                    "action": state["action"],
                }
            },
        )

    def apply_action_node(state: CollabGraphState) -> CollabGraphState:
        cmd = state["_cmd"]
        result = collab_service.apply_collab_action(cmd)
        return {
            **state,
            "result": result,
        }

    def finalize_rejected_node(state: CollabGraphState) -> CollabGraphState:
        return state

    graph.add_node("parse_command_node", parse_command_node)
    graph.add_node("risk_check_node", risk_check_node)
    graph.add_node("approval_node", approval_node)
    graph.add_node("apply_action_node", apply_action_node)
    graph.add_node("finalize_rejected_node", finalize_rejected_node)

    graph.add_edge(START, "parse_command_node")
    graph.add_edge("parse_command_node", "risk_check_node")
    graph.add_edge("risk_check_node", "approval_node")
    graph.add_edge("approval_node", "apply_action_node")
    graph.add_edge("apply_action_node", END)
    graph.add_edge("finalize_rejected_node", END)

    return graph.compile()
```
